Algorithmes fondamentaux appliqués aux mathématiques en Python

I. Algorithmes et fonctions

Étude d'une fonction
import math

def f(x):
    return x * math.exp(-x)

def derivee(f, x, h=1e-6):
    return (f(x + h) - f(x)) / h

# Table de valeurs
print("x\t\tf(x)\t\tf'(x)")
for x_val in [i * 0.5 for i in range(-4, 12)]:
    print(f"{x_val:.1f}\t\t{f(x_val):.4f}\t\t{derivee(f, x_val):.4f}")

II. Algorithmes et suites

Seuil de convergence
def seuil(u0, a, b, epsilon):
    """Trouve n tel que |u_n - limite| < epsilon"""
    limite = b / (1 - a)  # point fixe
    u = u0
    n = 0
    while abs(u - limite) >= epsilon:
        u = a * u + b
        n += 1
    return n, u

n, u = seuil(100, 0.8, 5, 0.01)
print(f"Convergence à 0.01 près en {n} étapes. u_n = {u:.4f}")
# Limite = 5/(1-0.8) = 25

III. Algorithmes et probabilités

Estimation de Pi par Monte Carlo
import random

def estimer_pi(n):
    dans_cercle = 0
    for _ in range(n):
        x = random.uniform(-1, 1)
        y = random.uniform(-1, 1)
        if x**2 + y**2 <= 1:
            dans_cercle += 1
    return 4 * dans_cercle / n

for n in [100, 1000, 10000, 100000]:
    pi_approx = estimer_pi(n)
    print(f"n = {n:>6} : pi ≈ {pi_approx:.4f}")

IV. Algorithmes et intégration

Méthode des trapèzes
def trapezes(f, a, b, n):
    h = (b - a) / n
    s = (f(a) + f(b)) / 2
    for i in range(1, n):
        s += f(a + i * h)
    return s * h

import math
# Intégrale de sin(x) de 0 à pi (valeur exacte = 2)
for n in [10, 100, 1000]:
    result = trapezes(math.sin, 0, math.pi, n)
    print(f"n = {n:>4} : ∫sin(x)dx ≈ {result:.6f} (erreur : {abs(result-2):.6f})")

V. Algorithmes et résolution d'équations

Méthode de dichotomie
def dichotomie(f, a, b, epsilon):
    """Trouve x tel que f(x) = 0 sur [a, b]"""
    while b - a > epsilon:
        m = (a + b) / 2
        if f(a) * f(m) <= 0:
            b = m
        else:
            a = m
    return (a + b) / 2

import math
# Résoudre e^x = x + 2
g = lambda x: math.exp(x) - x - 2
sol = dichotomie(g, -2, 0, 1e-8)
print(f"Solution : x ≈ {sol:.8f}")
sol2 = dichotomie(g, 1, 2, 1e-8)
print(f"Solution : x ≈ {sol2:.8f}")

Corrigé exercice 1

Corrigé
import math

def somme_inverses_carres(n):
    return sum(1 / k**2 for k in range(1, n + 1))

for n in [10, 100, 1000]:
    s = somme_inverses_carres(n)
    print(f"S({n}) = {s:.6f}  |  pi²/6 = {math.pi**2/6:.6f}  |  erreur = {abs(s - math.pi**2/6):.6f}")
# S(10) = 1.549768, S(100) = 1.634984, S(1000) = 1.643935
# pi²/6 = 1.644934. La somme converge lentement vers pi²/6.

Corrigé exercice 2

Corrigé
import math

def f(x):
    return math.log(x) - 1/x

solutions = 0
x = 0.1
pas = 0.001
prev = f(x)
while x <= 10:
    x += pas
    curr = f(x)
    if prev * curr < 0:  # changement de signe
        solutions += 1
        print(f"Solution trouvée vers x ≈ {x:.3f}")
    prev = curr

print(f"Nombre de solutions : {solutions}")
# Trouve 1 solution vers x ≈ 1.763
