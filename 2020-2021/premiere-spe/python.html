I. Python et fonctions

Définir et étudier une fonction
def f(x):
    return x**3 - 3*x + 1

# Table de valeurs
for x in range(-3, 4):
    print(f"f({x}) = {f(x)}")

# Recherche d'un zéro par balayage
x = -2.0
while f(x) < 0:
    x += 0.001
print(f"Zéro approché : x ≈ {x:.3f}")
Tangente numérique
def derivee_num(f, x, h=1e-6):
    return (f(x + h) - f(x)) / h

def tangente(f, x0, x):
    m = derivee_num(f, x0)
    return m * (x - x0) + f(x0)

# Vérifier que f'(1) = 0 pour f(x) = x³ - 3x + 1
print(f"f'(1) ≈ {derivee_num(f, 1):.6f}")  # ≈ 0

II. Python et suites

Générer une suite récurrente
def generer_suite(u0, relation, n):
    """relation est une fonction u -> u_suivant"""
    termes = [u0]
    for i in range(n):
        termes.append(relation(termes[-1]))
    return termes

# Suite u_{n+1} = 0.5*u_n + 3, u0 = 20
termes = generer_suite(20, lambda u: 0.5 * u + 3, 15)
for i, t in enumerate(termes):
    print(f"u_{i} = {t:.4f}")
# Converge vers 6 (point fixe)
Seuil de dépassement
# À partir de quel rang u_n > 1000 ?
# Suite géométrique u_n = 3 * 1.1^n
u = 3
n = 0
while u <= 1000:
    u = u * 1.1
    n += 1
print(f"u_n dépasse 1000 pour n = {n}")
# n = 61

III. Python et probabilités

Simuler un schéma de Bernoulli
import random

def experience_bernoulli(n, p):
    """Retourne le nombre de succès en n essais"""
    return sum(1 for _ in range(n) if random.random() < p)

# Simuler 1000 fois le lancer de 10 pièces
resultats = [experience_bernoulli(10, 0.5) for _ in range(1000)]

# Fréquence de chaque résultat
for k in range(11):
    freq = resultats.count(k) / 1000
    print(f"P(X={k}) ≈ {freq:.3f}")
Vérifier la loi des grands nombres
import random

p_vrai = 0.3
for n in [10, 100, 1000, 10000]:
    freq = sum(1 for _ in range(n) if random.random() < p_vrai) / n
    erreur = abs(freq - p_vrai)
    print(f"n = {n:>5} : freq = {freq:.4f}, erreur = {erreur:.4f}")
# L'erreur diminue quand n augmente

def harmonique(n):
    return sum(1/k for k in range(1, n + 1))

for n in [10, 100, 1000]:
    print(f"H({n}) = {harmonique(n):.4f}")
